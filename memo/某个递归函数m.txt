#include <iostream>
using namespace std;
double m(int i);
int main() {
    cout << "i\t\tm(i)\n";
    for(int i = 1; i <= 20; ++i)
        cout << i << "\t\t" << m(i) << endl;
    return 0;
}
double m(int i) {
    if (i == 1)
        return 1.0 / 2;
    else
        return m(i - 1) + i * 1.0 / (i + 1);
}
