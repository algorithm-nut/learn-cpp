/* 例4_13.cpp */
#include <iostream>
#include <array>
using namespace std;
int main() {
    array<int, 10> arr;
    cout << "数组当前有效元素数量: " << arr.size() << endl;
    cout << "数组可以容纳的最大元素数量: " << arr.max_size() << endl;
    for(int i = 0; i < 10; ++i)
        arr[i] = i;
    for(int i = 0; i < arr.size(); ++i)
        cout << arr[i] << " ";
    cout << endl;
    arr.fill(1);
    for(int i = 0; i < arr.size(); ++i)
        cout << arr[i] << " ";
    cout << endl;
    arr.front() = 100;
    arr[1] = 101;
    arr.at(2) = 102;
    arr.back() = 104;
    for(auto value : arr)
        cout << value << " ";
    cout << endl;
    return 0;
}
