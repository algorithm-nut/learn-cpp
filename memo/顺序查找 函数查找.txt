#include <iostream>
using namespace std;
template <typename T>
int linearSearch(T key, T list[], int arraySize);
template <typename T>
void printArray(T list[], int arraySize);
int main() {
    const int ARRAY_SIZE = 10;
    int list[ARRAY_SIZE] = {1, 4, 4, 2, 5, -3, 6, 2, 7, -8};
    printArray(list, ARRAY_SIZE);
    int key;
    cout << "输入需要查找的值: ";
    cin >> key;
    int index = linearSearch(key, list, ARRAY_SIZE);
    if(index == -1)
        cout << "查找失败!\n";
    else
        cout << "查找成功: " << key << "位于数组下标" << index << "位置\n";
    return 0;
}
template <typename T>
int linearSearch(T key, T list[], int arraySize) {
    for(int i = 0; i < arraySize; ++i) {
        if(key == list[i])
            return i;
    }
    return -1;
}
template <typename T>
void printArray(T list[], int arraySize) {
    for(int i = 0; i < arraySize; ++i)
        cout << list[i] << " ";
    cout << endl;
}
