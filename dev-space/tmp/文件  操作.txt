#include <stdio.h>
#include <stdlib.h>  //打开文件失败 
#define FILENAME1 "input.txt"
#define FILENAME2 "output.txt"

int main(void) {
	FILE *fp1, *fp2;
	int num = 0; 
	fp1 = fopen(FILENAME1,"r");
	fp2 = fopen(FILENAME2,"w");
	if(fp1==NULL) {
		printf("Failure to open the %s\n",FILENAME1);  //文件位于dev统计目录 
		exit(EXIT_FAILURE);
	} 
    while(1) {
    	fscanf(fp1,"%d",&num);  //会跳过空格，直接读取数字 
    	if(num == -1)  break;
    	fprintf(fp2,"%-4d",num);
    }
    fclose(fp1);
    fclose(fp2);
			
	return 0;	
}
