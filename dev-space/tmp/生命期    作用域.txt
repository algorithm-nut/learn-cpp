/* 例3_30.cpp */
#include <iostream>
using namespace std;
void t1(void);
void t2(void);
void t3(void);
int a = 2;
int main() {
    int a = 4;
    cout << "在main函数中, 局部变量a的值是" << a << endl;
    {
        int a = 6;
        cout << "在main函数复合语句中, 局部变量a的值是" << a << endl;
    }
    cout << "在main函数中, 局部变量a的值是" << a << endl;
    t1();
    t2();
    t3();
    t1();
    t2();
    t3();
    cout << "在main函数中, 局部变量a的值是" << a << endl;
    return 0;
}
void t1(void) {
    int a = 10;
    cout << "进入t1函数, 局部变量a的值是" << a << endl;
    a += 5;
    cout << "离开t1函数, 局部变量a的值是" << a << endl;
}
void t2(void) {
    static int a = 20;
    cout << "进入t2函数, 静态局部变量a的值是" << a << endl;
    a += 5;
    cout << "离开t2函数, 静态局部变量a的值是" << a << endl;
}
void t3(void) {
    cout << "进入t3函数, 全局变量a的值是" << a << endl;
    a += 5;
    cout << "离开t3函数, 全局变量a的值是" << a << endl;
}
